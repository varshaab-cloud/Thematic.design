    </div>
    <p style={{ fontSize: 13, color: "var(--base-color-gray-500, #909090)", margin: 0, lineHeight: 1.5 }}>Semantic purpose mapped to base values. Dark mode lives here.</p>

    '// tokens.json',

    '    }',
    '  }',
    '}',
    '',
    '// compiles to',
    '--color-brand-primary: var(--base-color-blue-800);',
  ]} />

    '  }',
